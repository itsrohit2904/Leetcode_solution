//509. Fibonacci Number
//using recursion..


class Solution {
    public int fib(int n) {
        if(n==1 || n==2){
            return 1;
        }
        if(n==0){
            return 0;
        }
        return fib(n-1)+fib(n-2);
    }
}


//optimized solution

class Solution {
    public int fib(int n) {
       int a=1;
       int b=0;
       int temp=0;
       while(n>0){
        temp=a;
        a=a+b;
        b=temp;
        n--;
       }
       return b;
    }
}
