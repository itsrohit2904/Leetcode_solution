//11. Container With Most Water


class Solution {
    public int maxArea(int[] height) {
        int start=0;
        int end=height.length-1;
        int max_value=0;
        while(start<end){
            max_value=Math.max(max_value,(end-start)*Math.min(height[start],height[end]));
            if(height[start]<height[end]){
                start++;
            }
            else{
                end--;
            }
        }
        return max_value;
        
    }
}
